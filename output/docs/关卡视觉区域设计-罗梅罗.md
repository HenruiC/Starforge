## 一、核心思路：Tile Atlas 扩展 + Source ID 分区

**当前问题（`scripts/school_map.gd`）：**
- 3 个 source ID：0=地板（floor.png）、1=墙壁（wall.png）、2=门（door.png）
- 所有区域共用同一套 tileset，视觉上全是深灰地板+灰墙
- `_draw()` 内部没有区域概念，全是全局 `for x in cols: for y in rows: set_cell(0, ...)`

**解决方案：给每个区域分配专属 source ID**

```
Source ID  用途                 纹理文件
0         操场地板 (浅沙色)     floor_playground.png
1         教室地板 (深灰)       floor_classroom.png
2         走廊地板 (浅灰带线)   floor_hallway.png
3         体育馆地板 (红褐色)    floor_gym.png
4         普通墙壁              wall_normal.png
5         教学楼内墙            wall_classroom.png
6         体育馆墙壁            wall_gym.png
7         门                    door.png
```

### 落地方式

在 `school_map.gd` 的 `_build_tileset()` 中（第 23-62 行），把 3 个 `_add_source()` 调用扩展为 8 个：


## 二、纹理生成方案（不需要美术出图）

现有 `AssetLoader.texture()` 已经支持按名称+颜色生成纯色纹。在 `_build_tileset()` 中不需要改文件路径，直接用代码生成：

### 在 `school_map.gd` 的 `_build_tileset()` 增加一个 `_add_color_source()` 方法

```gdscript
func _add_color_source(ts: TileSet, sid: int, color: Color) -> void:
    var img := Image.create(T, T, false, Image.FORMAT_RGBA8)
    img.fill(color)
    var tex := ImageTexture.create_from_image(img)
    var atlas := TileSetAtlasSource.new()
    atlas.texture = tex
    atlas.texture_region_size = Vector2i(T, T)
    atlas.create_tile(Vector2i(0, 0))
    ts.add_source(atlas, sid)
```

### 每个区域的配色方案

| 区域 | 地板色 | 墙壁色 | 视觉目的 |

### 2. ColorRect 背景块（当前 TileMap z_index=-5 之下）

在场景根节点加 4 个 ColorRect 作为区域泛光：

## 五、小怪反制远程 — 3 个具体实现

### 5.1 近战敌人：冲刺突进（"Shield Rush"）

**文件：`scripts/enemy.gd`**
**位置：`_melee_behavior()` 第 100-103 行**

当前逻辑：`melee` 只是追着玩家跑。改为：
- 当检测到玩家正在释放远程技能（通过 EventBus 接收 `skill_cast` 信号）时
- 近战敌人获得 0.5 秒 **速度翻倍 + 短暂无敌** 冲向你
- 视觉：sprite 变成半透明的 "冲刺残影" 状态

```gdscript
# 在 _melee_behavior() 追加这段（或在 _ready 中连接信号）：
func _on_skill_detected(skill_type: String, caster_pos: Vector2) -> void:
    if is_dead or _is_invincible or is_ranged:
        return
    var dist := global_position.distance_to(_player_ref.global_position)
    if dist > 300.0 or dist < 60.0:
        return  # 太远不冲，太近不用冲
    _trigger_rush()

func _trigger_rush() -> void:
    _is_invincible = true
    sprite.modulate = Color(1, 1, 1, 0.5)  # 半透明残影
    var rush_speed := move_speed * 2.5
    var dir := global_position.direction_to(_player_ref.global_position)
    velocity = dir * rush_speed
    move_and_slide()
    await get_tree().create_timer(0.5).timeout
    _is_invincible = false
    sprite.modulate = Color.WHITE
```

### 5.2 精英敌人：护盾减伤

**文件：`scripts/enemy.gd`**
**位置：`take_damage()` 第 136-152 行**

当前 `take_damage()` 只用 `_is_invincible` 做完全免疫。改为精英特有护盾减伤：

```gdscript
func take_damage(amount: int, source: CombatUnit = null) -> void:
    if is_dead: return
    if _is_invincible: return

    # 精英护盾：远程伤害减半
    var is_elite := is_elite or get_stat("is_elite", 0.0) > 0.5
    if is_elite and source and source.team == CombatUnit.Team.FRIENDLY:
        # 检查是否为远程攻击 — 由 source 传一个标志
        # 或者检查攻击者的阵营+距离组合
        var dist := global_position.distance_to(source.global_position)
        if dist > 80.0:  # 超过近战范围 = 远程攻击
            amount = maxi(amount / 2, 1)

    var actual := maxi(amount - int(get_stat("defense", 0.0)), 1)
    ...
```

同时给精英加一个间歇性护盾闪光提示：
```gdscript
# 在 _setup_elite() 第 52-66 行追加
# 每 5 秒闪一次蓝色护盾
var shield_tween := create_tween().set_loops()
shield_tween.tween_interval(5.0)
shield_tween.tween_callback(func():
    sprite.modulate = Color(0.3, 0.5, 1.0, 0.8)
    await get_tree().create_timer(0.15).timeout
    sprite.modulate = Color.WHITE
)
```

### 5.3 闪避型敌人（新类型）

**文件：`scripts/enemy.gd`**
**位置：新增 `is_dodger` 属性 + `_ranged_behavior()` 第 105-121 行 + `_physics_process()` 第 85-98 行**

在 `school_map.gd` 的 `get_preplaced_enemies()` 中（第 200-271 行），远程敌人已有 `"type": "ranged"`。新增 `"type": "dodger"` 类型：

```gdscript
# school_map.gd 追加：
{"pos": Vector2(1800, 2200), "type": "dodger", "activation": 280},
```

**enemy.gd 改动：**

1. 第 15 行追加导出属性：
```gdscript
@export var is_dodger: bool = false
@export var dodge_cooldown: float = 1.5
```

2. 第 85-98 行 `_physics_process()` 追加 dodger 分支：
```gdscript
if is_dodger:
    _dodger_behavior(delta)
```

3. 闪避实现：
```gdscript
var _dodge_timer: float = 0.0

func _dodger_behavior(delta: float) -> void:
    _dodge_timer += delta

    # 检查是否有弹幕接近
    var danger := _detect_incoming_projectile()
    if danger and _dodge_timer >= dodge_cooldown:
        _dodge(danger)
        return

    # 正常走 ranged_behavior 的逻辑
    _ranged_behavior(delta)

func _detect_incoming_projectile() -> bool:
    # 扫描附近 enemy_projectile（玩家弹幕）
    var projectiles := get_tree().get_nodes_in_group("player_projectile")
    for p in projectiles:
        var dist := global_position.distance_to(p.global_position)
        var dir_to_p := (p.global_position - global_position).normalized()
        # 弹幕朝自己来且在 100px 内
        if dist < 150.0:
            return true
    return false

func _dodge(from_pos: Vector2) -> void:
    _dodge_timer = 0.0
    # 侧向闪避 — 垂直于弹幕方向移动
    var danger_dir := (from_pos - global_position).normalized()
    var dodge_dir := danger_dir.rotated(PI / 2)  # 垂直方向
    if randf() > 0.5:
        dodge_dir = -dodge_dir  # 随机左右
    velocity = dodge_dir * move_speed * 2.0
    move_and_slide()

    # 视觉：快速闪烁
    var dodge_flash := create_tween()
    dodge_flash.tween_property(sprite, "modulate:a", 0.3, 0.08)
    dodge_flash.tween_property(sprite, "modulate:a", 1.0, 0.08)
```
